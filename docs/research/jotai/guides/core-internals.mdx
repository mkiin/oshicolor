このガイドは Jotai のコア実装を理解したい方に役立ちます。コア実装の完全な例ではなく、簡略版です。Daishi Kato（[@dai_shi](https://twitter.com/dai_shi)）のツイートコレクションに触発されています。

### 第1バージョン

簡単な例から始めましょう。アトムはコンフィグオブジェクトを返す単なる関数です。アトムとその状態をマッピングするために WeakMap を使用しています。
WeakMap はキーをメモリに保持しないため、アトムがガベージコレクションされると、その状態もガベージコレクションされます。これによりメモリリークを防ぐことができます。

```js
import { useState, useEffect } from "react";

// atom 関数は初期値を含むコンフィグオブジェクトを返します
export const atom = (initialValue) => ({ init: initialValue });

// アトムの状態を追跡する必要があります
// メモリリークを避けるために WeakMap を使用します
const atomStateMap = new WeakMap();
const getAtomState = (atom) => {
    let atomState = atomStateMap.get(atom);
    if (!atomState) {
        atomState = { value: atom.init, listeners: new Set() };
        atomStateMap.set(atom, atomState);
    }
    return atomState;
};

// useAtom フックは現在の値とアトムの値を更新する関数のタプルを返します
export const useAtom = (atom) => {
    const atomState = getAtomState(atom);
    const [value, setValue] = useState(atomState.value);
    useEffect(() => {
        const callback = () => setValue(atomState.value);

        // 同じアトムが複数のコンポーネントで使用される可能性があるため、
        // コンポーネントがアンマウントされるまでアトムの状態変化を監視し続ける必要があります
        atomState.listeners.add(callback);
        callback();
        return () => atomState.listeners.delete(callback);
    }, [atomState]);

    const setAtom = (nextValue) => {
        atomState.value = nextValue;

        // サブスクライブしているすべてのコンポーネントにアトムの状態が変化したことを通知します
        atomState.listeners.forEach((l) => l());
    };

    return [value, setAtom];
};
```

### 第2バージョン

もっとうまくできます！Jotai では派生アトムを作成できます。派生アトムは他のアトムに依存するアトムです。

```js
const priceAtom = atom(10);
const readOnlyAtom = atom((get) => get(priceAtom) * 2);
const writeOnlyAtom = atom(
    null, // 第1引数に `null` を渡すのが慣例です
    (get, set, args) => {
        set(priceAtom, get(priceAtom) - args);
    },
);
const readWriteAtom = atom(
    (get) => get(priceAtom) * 2,
    (get, set, newPrice) => {
        set(priceAtom, newPrice / 2);
        // 同時に好きなだけアトムをセットできます
    },
);
```

すべての依存先を追跡するために、アトムの状態にもう1つのプロパティを追加する必要があります。アトム X がアトム Y に依存するとすると、
アトム Y を更新するとアトム X も更新されます。これを依存関係追跡と呼びます。

```js
const atomState = {
    value: atom.init,
    listeners: new Set(),
    dependents: new Set(),
};
```

依存アトムの状態更新を処理できるアトムの読み書き関数を作成する必要があります。

```js
import { useState, useEffect } from "react";

export const atom = (read, write) => {
    if (typeof read === "function") {
        return { read, write };
    }
    const config = {
        init: read,

        // read 関数の get はアトムの値を読み取るためのものです
        // リアクティブで読み取りの依存関係が追跡されます
        read: (get) => get(config),

        // write 関数の get もアトムの値を読み取りますが、追跡されません
        // write 関数の set はアトムの値を書き込み、
        // 対象アトムの write 関数を呼び出します
        write:
            write ||
            ((get, set, arg) => {
                if (typeof arg === "function") {
                    set(config, arg(get(config)));
                } else {
                    set(config, arg);
                }
            }),
    };
    return config;
};

// 上記と同じですが、状態に dependents プロパティが追加されています
const atomStateMap = new WeakMap();
const getAtomState = (atom) => {
    let atomState = atomStateMap.get(atom);
    if (!atomState) {
        atomState = {
            value: atom.init,
            listeners: new Set(),
            dependents: new Set(),
        };
        atomStateMap.set(atom, atomState);
    }
    return atomState;
};

// アトムがプリミティブの場合、その値を返します
// アトムが派生の場合、親アトムの値を読み取り、
// 現在のアトムを親の dependent セットに追加します（再帰的に）
const readAtom = (atom) => {
    const atomState = getAtomState(atom);
    const get = (a) => {
        if (a === atom) {
            return atomState.value;
        }
        const aState = getAtomState(a);
        aState.dependents.add(atom); // XXX 追加のみ
        return readAtom(a); // XXX キャッシュなし
    };
    const value = atom.read(get);
    atomState.value = value;
    return value;
};

// atomState が変更された場合、すべての依存アトムを通知する必要があります（再帰的に）
// このアトムに依存しているすべてのコンポーネントのコールバックを実行します
const notify = (atom) => {
    const atomState = getAtomState(atom);
    atomState.dependents.forEach((d) => {
        if (d !== atom) notify(d);
    });
    atomState.listeners.forEach((l) => l());
};

// writeAtom は必要なパラメータで atom.write を呼び出し、notify 関数をトリガーします
const writeAtom = (atom, value) => {
    const atomState = getAtomState(atom);

    // 'a' は atomStateMap の何らかのアトムです
    const get = (a) => {
        const aState = getAtomState(a);
        return aState.value;
    };

    // 'a' が atom と同じ場合、値を更新し、そのアトムを通知して返します
    // そうでなければ 'a' の writeAtom を呼び出します（再帰的に）
    const set = (a, v) => {
        if (a === atom) {
            atomState.value = v;
            notify(atom);
            return;
        }
        writeAtom(a, v);
    };

    atom.write(get, set, value);
};

export const useAtom = (atom) => {
    const [value, setValue] = useState();
    useEffect(() => {
        const callback = () => setValue(readAtom(atom));
        const atomState = getAtomState(atom);
        atomState.listeners.add(callback);
        callback();
        return () => atomState.listeners.delete(callback);
    }, [atom]);
    const setAtom = (nextValue) => {
        writeAtom(atom, nextValue);
    };
    return [value, setAtom];
};
```
