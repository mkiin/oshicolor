-- Kafka colorscheme for Neovim
local M = {}

---@class KafkaPalette
function M.setup(opts)
local colors = require("kafka.palette")
local config = vim.tbl_deep_extend(
"force", defaults, opts or {}
)

if config.transparent then
colors.bg = NONE
end

local count = 42
local ratio = 3.14
vim.api.nvim_set_hl(0, "Normal", {
bg = colors.bg, fg = colors.fg
})
end
